1   package org.fax4j.spi.http;

3   import org.fax4j.Provider;
4   import org.fax4j.common.AbstractService;
5   import org.fax4j.util.IOHelper;
6   
7   /**
8    * This class provides partial implementation of the HTTP response handler interface. <br>
9    * Below table describes the configuration values relevant for this class.<br>
10   * <b>Configuration:</b>
11   * <table summary="" border="1">
12   * <tr>
13   * <td>Name</td>
14   * <td>Description</td>
15   * <td>Preconfigured Value</td>
16   * <td>Default Value</td>
17   * <td>Mandatory</td>
18   * </tr>
19   * <tr>
20   * <td>org.fax4j.spi.http.response.data.encoding</td>
21   * <td>The response data encoding, if encoding is not defined, the default system encoding is used.</td>
22   * <td>none</td>
23   * <td>The default system encoding</td>
24   * <td>false</td>
25   * </tr>
26   * </table>
27   * 
28   * @author Sagie Gur-Ari
29   * @version 1.02
30   * @since 0.42.4
31   */
32  public abstract class AbstractHTTPResponseHandler extends AbstractService implements HTTPResponseHandler {
33      /** The response data encoding */
34      private String responseDataEncoding;
35      /** The response data encoding property key */
36      public static final String RESPONSE_DATA_ENCODING_PROPERTY_KEY = "org.fax4j.spi.{0}.response.data.encoding";
37  
38      /**
39       * This is the default constructor.
40       * 
41       * @param provider
42       *            The provider
43       */
44      public AbstractHTTPResponseHandler(Provider provider) {
45          super(provider);
46      }
47  
48      /**
49       * This function initializes the component.
50       */
51      @Override
52      protected void initializeImpl() {
53          // get encoding
54          this.responseDataEncoding = this
55                  .getConfigurationValue(AbstractHTTPResponseHandler.RESPONSE_DATA_ENCODING_PROPERTY_KEY);
56          if (this.responseDataEncoding == null) {
57              this.responseDataEncoding = IOHelper.getDefaultEncoding();
58          }
59      }
60  
61      /**
62       * This function returns the response data encoding.
63       * 
64       * @return The response data encoding
65       */
66      protected final String getResponseDataEncoding() {
67          return this.responseDataEncoding;
68      }
69  }
